select l_shipdate, sum(l_extendedprice*(1-l_discount)/1000) as revenue from supplier, lineitem, orders, customer, ids as id1, ids as id2  where s_suppkey = l_suppkey and o_orderkey = l_orderkey and c_custkey = o_custkey and id1.i_id = supplier.s_id and id2.i_id = customer.c_id and l_shipdate between date '1995-01-01' and date '1995-4-10' group by l_shipdate;
